    <script>
        // Tailwind initialization
        function initializeTailwind() {

                }
            }
            return true
        }
        
        // Global variables
        let linkBudgetChart = null
        let pathLossChart = null
        let elevationChart = null
        
        // Orbit animation globals
        let canvas, ctx
        let animationFrame = null
        let isOrbitPlaying = false
        let orbitTime = 0
        let currentOrbitType = 'LEO'
        let orbitSpeed = 1.0
        let elevationHistory = []
        let elevationTimeLabels = []
        
        // Real orbital parameters (km)
        const orbitParams = {
            LEO: { name: 'LEO', h_km: 800, r_km: 7171, period_min: 100, color: '#67e8f9' },
            MEO: { name: 'MEO', h_km: 20200, r_km: 26571, period_min: 720, color: '#a78bfa' },
            GEO: { name: 'GEO', h_km: 35786, r_km: 42157, period_min: 1440, color: '#f472b6' }
        }
        
        // Display radii (exaggerated for visibility)
        const displayRadii = { LEO: 215, MEO: 340, GEO: 465 }
        
        // Bands for path loss
        const frequencyBands = [
            { name: 'L', f: 1.5, color: '#67e8f9' },
            { name: 'S', f: 3, color: '#a3e635' },
            { name: 'C', f: 6, color: '#22d3ee' },
            { name: 'X', f: 10, color: '#6366f1' },
            { name: 'Ka', f: 30, color: '#c026d3' }
        ]
        
        // Link budget shared inputs
        let currentInputs = {
            pt: 20,
            gt: 25,
            gr: 35,
            freq: 12,
            loss: 4,
            sens: -115
        }
        
        // FSPL helper
        function calculateFSPL(distance_km, freq_ghz) {
            return 92.45 + 20 * Math.log10(distance_km) + 20 * Math.log10(freq_ghz)
        }
        
        // Calculate full budget for one orbit
        function computeOrbitBudget(orbitKey) {
            const params = orbitParams[orbitKey]
            const d = params.h_km   // overhead slant range = altitude
            const fspl = calculateFSPL(d, currentInputs.freq)
            const pr = currentInputs.pt + currentInputs.gt + currentInputs.gr - fspl - currentInputs.loss
            const margin = pr - currentInputs.sens
            return {
                name: params.name,
                h: params.h_km,
                d: Math.round(d),
                fspl: fspl.toFixed(1),
                pr: pr.toFixed(1),
                margin: margin.toFixed(1),
                marginNum: margin
            }
        }
        
        // Render link budget table + chart
        function calculateLinkBudget() {
            // Read current values
            currentInputs.pt = parseFloat(document.getElementById('pt').value)
            currentInputs.gt = parseFloat(document.getElementById('gt').value)
            currentInputs.gr = parseFloat(document.getElementById('gr').value)
            currentInputs.freq = parseFloat(document.getElementById('freq').value)
            currentInputs.loss = parseFloat(document.getElementById('loss').value)
            currentInputs.sens = parseFloat(document.getElementById('sens').value)
            
            // EIRP
            const eirp = currentInputs.pt + currentInputs.gt
            document.getElementById('eirp-display').textContent = eirp.toFixed(1) + ' dBW'
            
            // Table
            const tbody = document.getElementById('budget-table')
            tbody.innerHTML = ''
            
            const results = {}
            const marginData = []
            const labels = []
            
            Object.keys(orbitParams).forEach(key => {
                const res = computeOrbitBudget(key)
                results[key] = res
                
                const rowHTML = `
                <tr class="border-b border-white/10 last:border-none hover:bg-white/5 transition-colors">
                    <td class="py-6 font-semibold">${res.name}</td>
                    <td class="py-6">${res.h}</td>
                    <td class="py-6">${res.d}</td>
                    <td class="py-6 text-amber-300">${res.fspl}</td>
                    <td class="py-6 font-semibold">${res.pr}</td>
                    <td class="py-6 ${res.marginNum >= 0 ? 'text-emerald-400' : 'text-red-400'}">${res.margin}</td>
                    <td class="py-6">
                        ${res.marginNum >= 0 
                            ? `<span class="inline-flex items-center text-xs px-3 py-1 bg-emerald-400/10 text-emerald-400 rounded-2xl">✅ LINK OK</span>` 
                            : `<span class="inline-flex items-center text-xs px-3 py-1 bg-red-400/10 text-red-400 rounded-2xl">⚠️ MARGIN LOW</span>`}
                    </td>
                </tr>`
                tbody.innerHTML += rowHTML
                
                marginData.push(res.marginNum)
                labels.push(res.name)
            })
            
            // Update or create bar chart
            if (!linkBudgetChart) {
                const ctx = document.getElementById('margin-chart')
                linkBudgetChart = new Chart(ctx, {
                    type: 'bar',
                    data: {
                        labels: labels,
                        datasets: [{
                            label: 'Link Margin (dB)',
                            data: marginData,
                            backgroundColor: ['#67e8f9', '#a78bfa', '#f472b6'],
                            borderColor: '#ffffff',
                            borderWidth: 2,
                            borderRadius: 8,
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: { legend: { display: false } },
                        scales: {
                            y: { grid: { color: '#334155' }, ticks: { color: '#e0f2fe' } },
                            x: { grid: { display: false }, ticks: { color: '#e0f2fe' } }
                        }
                    }
                })
            } else {
                linkBudgetChart.data.datasets[0].data = marginData
                linkBudgetChart.update('none')
            }
        }
        
        // Sync number <-> slider
        function updateInputValue(slider, id) {
            document.getElementById(id).value = slider.value
            calculateLinkBudget()
        }
        
        function syncSlider(numInput, sliderId) {
            document.getElementById(sliderId).value = numInput.value
            calculateLinkBudget()
        }
        
        function setFreq(val) {
            document.getElementById('freq').value = val
            calculateLinkBudget()
        }
        
        // Path loss chart
        function createPathLossChart() {
            const ctx = document.getElementById('path-chart')
            
            // Log-spaced distances
            const distances = []
            for (let i = 2; i <= 5.01; i += 0.05) {
                distances.push(Math.pow(10, i))
            }
            
            const datasets = frequencyBands.map(band => {
                return {
                    label: band.name + '-band',
                    data: distances.map(d => calculateFSPL(d, band.f)),
                    borderColor: band.color,
                    borderWidth: 3,
                    tension: 0.1,
                    pointRadius: 0
                }
            })
            
            pathLossChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: distances,
                    datasets: datasets
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,

                                label: (ctx) => ctx.raw.toFixed(1) + ' dB',

                            grid: { color: '#334155' },
                            title: { display: true, text: 'Free Space Path Loss (dB)', color: '#e0f2fe' }
                        }
                    }
                }
            })
            
            // Populate instant loss grid once
            updateInstantLossGrid(1000)
        }

        // Orbit canvas drawing
        function drawOrbit() {
            if (!ctx) return
            
            ctx.clearRect(0, 0, canvas.width, canvas.height)
            
            const centerX = canvas.width / 2
            const centerY = canvas.height / 2 - 10
            const earthPx = 102
            const params = orbitParams[currentOrbitType]
            const displayR = displayRadii[currentOrbitType]
            
            // Earth
            ctx.shadowColor = '#67e8f9'
            ctx.shadowBlur = 30
            ctx.fillStyle = '#1e40af'
            ctx.beginPath()
            ctx.arc(centerX, centerY, earthPx, 0, Math.PI * 2)
            ctx.fill()
            ctx.shadowBlur = 0
            
            // Simple earth glow
            ctx.strokeStyle = '#67e8f9'
            ctx.lineWidth = 8
            ctx.globalAlpha = 0.15
            ctx.beginPath()
            ctx.arc(centerX, centerY, earthPx + 12, 0, Math.PI * 2)
            ctx.stroke()
            ctx.globalAlpha = 1.0
            
            // Orbit ring
            ctx.strokeStyle = '#64748b'
            ctx.lineWidth = 4
            ctx.setLineDash([2, 6])
            ctx.beginPath()
            ctx.arc(centerX, centerY, displayR, 0, Math.PI * 2)
            ctx.stroke()
            ctx.setLineDash([])
            
            // Calculate real physics angles
            const realRatio = params.r_km / 6371
            const realBeta = Math.acos(1 / realRatio)   // radians
            
            // Satellite angle (radians)
            const theta = orbitTime * orbitSpeed * (2 * Math.PI / (params.period_min * 0.8))   // normalized speed
            
            // Satellite position (display)
            const satX = centerX + displayR * Math.cos(theta)
            const satY = centerY + displayR * Math.sin(theta)
            
            // Draw satellite
            ctx.save()
            ctx.translate(satX, satY)
            ctx.rotate(theta + Math.PI / 4)
            ctx.fillStyle = params.color
            ctx.shadowColor = params.color
            ctx.shadowBlur = 25
            ctx.fillRect(-14, -7, 28, 14)   // satellite body
            ctx.fillStyle = '#f1f5f9'
            ctx.fillRect(10, -3, 12, 6)     // solar panel
            ctx.restore()
            
            // Nadir angle = theta
            const nadirAngle = theta
            
            // Footprint arc (real physics beta)
            const t1 = nadirAngle - realBeta
            const t2 = nadirAngle + realBeta
            
            // Highlighted footprint band on Earth surface
            ctx.save()
            ctx.fillStyle = 'rgba(103, 232, 249, 0.25)'
            ctx.beginPath()
            ctx.arc(centerX, centerY, earthPx, t1, t2)
            ctx.arc(centerX, centerY, earthPx + 14, t2, t1, true)
            ctx.closePath()
            ctx.fill()
            ctx.restore()
            
            // Tangent beams (visual)
            const tx1x = centerX + earthPx * Math.cos(t1)
            const tx1y = centerY + earthPx * Math.sin(t1)
            const tx2x = centerX + earthPx * Math.cos(t2)
            const tx2y = centerY + earthPx * Math.sin(t2)
            
            ctx.strokeStyle = '#67e8f9'
            ctx.lineWidth = 2.5
            ctx.globalAlpha = 0.6
            ctx.beginPath()
            ctx.moveTo(satX, satY)
            ctx.lineTo(tx1x, tx1y)
            ctx.moveTo(satX, satY)
            ctx.lineTo(tx2x, tx2y)
            ctx.stroke()
            ctx.globalAlpha = 1.0
            
            // Ground station (fixed at bottom)
            const gsAngle = Math.PI * 1.4   // slightly offset for nice view
            const gsX = centerX + earthPx * Math.cos(gsAngle)
            const gsY = centerY + earthPx * Math.sin(gsAngle)
            ctx.fillStyle = '#f43f5e'
            ctx.beginPath()
            ctx.arc(gsX, gsY, 9, 0, Math.PI * 2)
            ctx.fill()
            ctx.fillStyle = '#fff'
            ctx.font = '10px monospace'
            ctx.fillText('GS', gsX - 8, gsY + 3)
            
            // Link line when visible
            const phi = Math.min(Math.abs(theta - gsAngle), 2 * Math.PI - Math.abs(theta - gsAngle))
            const isVisible = phi <= realBeta
            if (isVisible) {
                ctx.strokeStyle = '#67e8f9'
                ctx.lineWidth = 3
                ctx.setLineDash([1, 2])
                ctx.beginPath()
                ctx.moveTo(satX, satY)
                ctx.lineTo(gsX, gsY)
                ctx.stroke()
                ctx.setLineDash([])
            }
            
            // Elevation calculation (real)
            const Re = 6371
            const r = params.r_km
            const d_slant = Math.sqrt(r * r + Re * Re - 2 * r * Re * Math.cos(phi))
            let elevation = -90
            if (d_slant > 0) {
                elevation = Math.asin((r * Math.cos(phi) - Re) / d_slant) * (180 / Math.PI)
            }
            elevation = Math.max(-10, Math.min(90, elevation))
            
            // Update overlay
            document.getElementById('overlay-orbit').textContent = params.name
            document.getElementById('overlay-alt').textContent = params.h_km + ' km'
            document.getElementById('overlay-period').textContent = '~' + params.period_min + ' min'
            document.getElementById('overlay-el').textContent = elevation.toFixed(0) + '°'
            document.getElementById('footprint-size').textContent = (realBeta * 2 * 180 / Math.PI).toFixed(0) + '°'
            document.getElementById('slant-range').textContent = Math.round(d_slant) + ' km'
            document.getElementById('phi-angle').textContent = (phi * 180 / Math.PI).toFixed(0) + '°'
            
            // GEO special note
            document.getElementById('geo-note').classList.toggle('hidden', currentOrbitType !== 'GEO')
            
            // Record for elevation chart
            elevationHistory.push(elevation)
            elevationTimeLabels.push((orbitTime * 0.1).toFixed(1))
            if (elevationHistory.length > 80) {
                elevationHistory.shift()
                elevationTimeLabels.shift()
            }
            
            // Update elevation chart every 8 frames
            if (Math.random() < 0.12 && elevationChart) {
                elevationChart.data.labels = elevationTimeLabels
                elevationChart.data.datasets[0].data = elevationHistory
                elevationChart.update('none')
            }
            
            // Continue animation
            if (isOrbitPlaying) {
                orbitTime += 0.016 * orbitSpeed * 1.8
                animationFrame = requestAnimationFrame(drawOrbit)
            }
        }
        
        function selectOrbit(type) {
            currentOrbitType = type
            // Reset buttons
            document.querySelectorAll('.orbit-btn').forEach(b => {
                b.dataset.active = (b.id === `orbit-${type.toLowerCase()}-btn`)
            })
            if (!isOrbitPlaying) drawOrbit()   // redraw static
        }
        
        function toggleOrbitAnimation() {
            isOrbitPlaying = !isOrbitPlaying
            const textEl = document.getElementById('orbit-play-text')
            const statusEl = document.getElementById('orbit-status')
            
            if (isOrbitPlaying) {
                textEl.textContent = '❚❚'
                statusEl.textContent = 'PAUSE'
                orbitTime = 0   // restart for clean loop
                drawOrbit()
            } else {
                textEl.textContent = '▶'
                statusEl.textContent = 'PLAY ANIMATION'
                cancelAnimationFrame(animationFrame)
            }
        }

        // Tab system
        function showTab(n) {
            document.querySelectorAll('.tab-panel').forEach(p => p.classList.add('hidden'))
            document.getElementById('panel-' + n).classList.remove('hidden')
            
            // Activate tab button
            document.querySelectorAll('.tab-button').forEach((b, i) => {
                if (i === n) b.classList.add('tab-active')
                else b.classList.remove('tab-active')
            })
            
            // Lazy init charts when tab shown
            if (n === 0 && !linkBudgetChart) calculateLinkBudget()
            if (n === 1 && !pathLossChart) createPathLossChart()
            if (n === 2 && !elevationChart) {
                createElevationChart()
                // Boot canvas
                canvas = document.getElementById('orbit-canvas')
                ctx = canvas.getContext('2d')
                selectOrbit('LEO')
                drawOrbit()
            }
        }
